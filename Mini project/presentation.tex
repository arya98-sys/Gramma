% Slide 3: Introduction
\section{Project Overview}
\begin{frame}{What is Gramma AI?}
    \begin{itemize}
        \item \textbf{Definition:} A sophisticated, neural-network-based writing assistant designed for the modern web.
        \item \textbf{Neural Engine:} Utilizes the \textbf{T5 (Text-to-Text Transfer Transformer)} model for sequence-to-sequence grammar correction.
        \item \textbf{v2.0 Design Philosophy:} Features a bespoke "Dark Silk" interface focused on luxury aesthetics and user focus (Glassmorphism).
        \item \textbf{Key Value:} Bridges the gap between casual drafts and professional-grade communication by correcting syntax, semantics, and tone simultaneously.
    \end{itemize}
\end{frame}

% Slide 4: Problem Statement
\begin{frame}{The Problem Statement}
    \begin{block}{The Industry Challenge}
        Traditional rule-based spell-checkers focus on spelling but ignore context, resulting in:
    \end{block}
    \begin{itemize}
        \item \textbf{Contextual Blindness:} Grammatically "correct" sentences that feel awkward or unnatural to native speakers.
        \item \textbf{Tone Inconsistency:} Inability to distinguish between a formal business email and a casual text message.
        \item \textbf{UI Fatigue:} Standard tools often utilize cluttered interfaces that distract the user from the creative writing process.
        \item \textbf{Build Instability:} Development challenges in modernizing legacy React tools for high-performance AI integration.
    \end{itemize}
\end{frame}

% Slide 5: System Architecture
\section{System Architecture}
\begin{frame}{Technical Architecture \& Tech Stack}
    \begin{itemize}
        \item \textbf{Frontend UI:} React.js v18 utilizing \textbf{Tailwind CSS v3} for the "Dark Silk" glassmorphism effect and responsive design.
        \item \textbf{API Layer:} Python-based \textbf{FastAPI} ensuring high-concurrency and sub-second latency for AI inference.
        \item \textbf{AI Model:} Integration of \texttt{HappyTransformer} to deploy the T5 model, specifically fine-tuned for GEC (Grammar Error Correction).
        \item \textbf{Version Control:} Standardized Git-based collaborative workflow using a strict \textbf{Pull-Add-Commit-Push} protocol to maintain repository integrity.
    \end{itemize}
\end{frame}

% Slide 6: Data Workflow Architecture
\section{Data Workflow}
\begin{frame}{Project Data Workflow}
    \begin{itemize}
        \item \textbf{Phase 1: Client-Side Capture}
        \begin{itemize}
            \item User inputs "raw" text into the React textarea.
            \item Frontend performs basic validation (character count, empty check).
            \item Axios sends an asynchronous POST request with a JSON payload.
        \end{itemize}
        
        \item \textbf{Phase 2: API Orchestration}
        \begin{itemize}
            \item FastAPI receives the JSON object and parses the string.
            \item The request is queued for the HappyTransformer inference engine.
        \end{itemize}
        
        \item \textbf{Phase 3: Neural Processing (The T5 Core)}
        \begin{itemize}
            \item Text is tokenized into numerical vectors.
            \item The Transformer model performs a sequence-to-sequence map to predict the "corrected" version.
            \item Output is de-tokenized back into human-readable string format.
        \end{itemize}
        
        \item \textbf{Phase 4: State Update}
        \begin{itemize}
            \item FastAPI returns the refined text as a JSON response.
            \item React updates the application state, triggering the "Silk-UI" result card animation.
        \end{itemize}
    \end{itemize}
\end{frame}
